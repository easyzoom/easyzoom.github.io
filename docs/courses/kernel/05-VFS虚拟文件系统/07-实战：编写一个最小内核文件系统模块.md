# 实战：编写一个最小内核文件系统模块

## 前言

**C：** 前面六篇讲了 VFS 的概念、数据结构、路径查找、挂载、页缓存。理解它们最好的方式，是自己**写一个文件系统**——哪怕是最简单的那种。这一篇我们从零实现一个内存文件系统 `simplefs`：它能注册到 VFS、挂载、创建文件和目录、读写数据。整个模块不到 300 行 C 代码，但覆盖了 VFS 四大对象的核心回调。

<!-- more -->

## 目标

实现一个名为 `simplefs` 的内核文件系统模块，支持：

- `mount -t simplefs none /mnt/test`
- 在挂载点下创建文件和目录
- 读写文件内容（数据存在内存里）
- `ls`、`cat`、`echo >` 等基本操作
- `umount /mnt/test`

不支持（为了保持简单）：持久化、硬链接、符号链接、权限修改。

## 准备工作

```bash
# 确保安装了内核头文件
sudo apt install linux-headers-$(uname -r) build-essential

# 创建工作目录
mkdir ~/simplefs && cd ~/simplefs
```

## 完整代码

### simplefs.c

```c
#include <linux/module.h>
#include <linux/fs.h>
#include <linux/pagemap.h>
#include <linux/init.h>
#include <linux/namei.h>
#include <linux/slab.h>

#define SIMPLEFS_MAGIC 0x20260422

static const struct super_operations simplefs_s_ops;
static const struct inode_operations simplefs_dir_inode_ops;
static const struct file_operations simplefs_dir_ops;
static const struct file_operations simplefs_file_ops;
static const struct inode_operations simplefs_file_inode_ops;
static const struct address_space_operations simplefs_aops;

/*
 * 创建一个新的 VFS inode。
 * 这里不涉及磁盘，所有数据都在内存里。
 */
static struct inode *simplefs_make_inode(struct super_block *sb,
                                          umode_t mode)
{
    struct inode *inode = new_inode(sb);
    if (!inode)
        return NULL;

    inode->i_ino = get_next_ino();
    inode->i_mode = mode;
    simple_inode_init_ts(inode);
    inode->i_mapping->a_ops = &simplefs_aops;

    if (S_ISDIR(mode)) {
        inode->i_op = &simplefs_dir_inode_ops;
        inode->i_fop = &simplefs_dir_ops;
        inc_nlink(inode);
    } else if (S_ISREG(mode)) {
        inode->i_op = &simplefs_file_inode_ops;
        inode->i_fop = &simplefs_file_ops;
    }

    return inode;
}

/*
 * 在目录下创建一个新文件。
 * 对应 inode_operations.create 回调。
 */
static int simplefs_create(struct mnt_idmap *idmap,
                            struct inode *dir,
                            struct dentry *dentry,
                            umode_t mode, bool excl)
{
    struct inode *inode = simplefs_make_inode(dir->i_sb,
                                              mode | S_IFREG);
    if (!inode)
        return -ENOMEM;

    d_instantiate(dentry, inode);
    dget(dentry);
    dir->__i_mtime = dir->__i_ctime = current_time(dir);
    return 0;
}

/*
 * 在目录下创建一个子目录。
 * 对应 inode_operations.mkdir 回调。
 */
static int simplefs_mkdir(struct mnt_idmap *idmap,
                           struct inode *dir,
                           struct dentry *dentry,
                           umode_t mode)
{
    struct inode *inode = simplefs_make_inode(dir->i_sb,
                                              mode | S_IFDIR);
    if (!inode)
        return -ENOMEM;

    d_instantiate(dentry, inode);
    dget(dentry);
    inc_nlink(dir);
    dir->__i_mtime = dir->__i_ctime = current_time(dir);
    return 0;
}

/*
 * 在目录中查找一个名字。
 * 我们的 FS 完全靠 dcache，所以 lookup 只需要
 * 返回一个负 dentry（表示"我不知道"，由 VFS 处理）。
 */
static struct dentry *simplefs_lookup(struct inode *dir,
                                       struct dentry *dentry,
                                       unsigned int flags)
{
    return simple_lookup(dir, dentry, flags);
}

/* 目录 inode 操作表 */
static const struct inode_operations simplefs_dir_inode_ops = {
    .lookup = simplefs_lookup,
    .create = simplefs_create,
    .mkdir  = simplefs_mkdir,
};

/* 目录 file 操作表：用内核提供的简单目录迭代 */
static const struct file_operations simplefs_dir_ops = {
    .open       = dcache_dir_open,
    .release    = dcache_dir_close,
    .llseek     = dcache_dir_lseek,
    .read       = generic_read_dir,
    .iterate_shared = dcache_readdir,
};

/*
 * 文件的 address_space_operations。
 * 使用内核提供的 simple_* 系列函数，
 * 数据直接存在页缓存里（内存 FS）。
 */
static const struct address_space_operations simplefs_aops = {
    .readpage    = simple_readpage,
    .write_begin = simple_write_begin,
    .write_end   = simple_write_end,
};

/*
 * 文件的 file_operations。
 * 使用 generic 系列——对于内存 FS 这就够了。
 */
static const struct file_operations simplefs_file_ops = {
    .llseek     = generic_file_llseek,
    .read_iter  = generic_file_read_iter,
    .write_iter = generic_file_write_iter,
    .mmap       = generic_file_mmap,
    .fsync      = noop_fsync,
};

/* 文件的 inode_operations (简单实现只需 getattr/setattr) */
static const struct inode_operations simplefs_file_inode_ops = {
    .getattr = simple_getattr,
    .setattr = simple_setattr,
};

/* super_operations */
static const struct super_operations simplefs_s_ops = {
    .statfs     = simple_statfs,
    .drop_inode = generic_delete_inode,
};

/*
 * 填充超级块：设置魔数、块大小、根目录。
 */
static int simplefs_fill_super(struct super_block *sb, void *data,
                                int silent)
{
    struct inode *root_inode;

    sb->s_magic = SIMPLEFS_MAGIC;
    sb->s_blocksize = PAGE_SIZE;
    sb->s_blocksize_bits = PAGE_SHIFT;
    sb->s_maxbytes = MAX_LFS_FILESIZE;
    sb->s_op = &simplefs_s_ops;
    sb->s_time_gran = 1;

    root_inode = simplefs_make_inode(sb, S_IFDIR | 0755);
    if (!root_inode)
        return -ENOMEM;

    sb->s_root = d_make_root(root_inode);
    if (!sb->s_root)
        return -ENOMEM;

    return 0;
}

/*
 * mount 回调。
 * 使用 mount_nodev——不需要块设备。
 */
static struct dentry *simplefs_mount(struct file_system_type *fs_type,
                                      int flags,
                                      const char *dev_name,
                                      void *data)
{
    return mount_nodev(fs_type, flags, data, simplefs_fill_super);
}

/* 文件系统类型 */
static struct file_system_type simplefs_type = {
    .owner   = THIS_MODULE,
    .name    = "simplefs",
    .mount   = simplefs_mount,
    .kill_sb = kill_litter_super,
};

static int __init simplefs_init(void)
{
    int ret = register_filesystem(&simplefs_type);
    if (ret == 0)
        pr_info("simplefs: registered\n");
    return ret;
}

static void __exit simplefs_exit(void)
{
    unregister_filesystem(&simplefs_type);
    pr_info("simplefs: unregistered\n");
}

module_init(simplefs_init);
module_exit(simplefs_exit);

MODULE_LICENSE("GPL");
MODULE_AUTHOR("EASYZOOM");
MODULE_DESCRIPTION("A minimal in-memory filesystem for learning VFS");
```

## 代码解析：对应 VFS 概念

### 注册文件系统

```c
static struct file_system_type simplefs_type = {
    .name    = "simplefs",
    .mount   = simplefs_mount,
    .kill_sb = kill_litter_super,
};
register_filesystem(&simplefs_type);
```

这一步让 VFS 知道"世界上有一个叫 simplefs 的文件系统"。`/proc/filesystems` 里就能看到了。

### mount 回调 → 填充 super_block

`simplefs_mount()` 调用 `mount_nodev()`（因为我们不需要块设备），最终 VFS 会回调 `simplefs_fill_super()` 来填充 `struct super_block`：

| 我们做的 | 对应 VFS 概念 |
|----------|---------------|
| `sb->s_op = &simplefs_s_ops` | 注册 super_operations |
| `simplefs_make_inode(sb, S_IFDIR)` | 创建根目录的 inode |
| `d_make_root(root_inode)` | 创建根 dentry 并关联 inode |
| `sb->s_root = ...` | 设置这个 FS 的根 dentry |

### 创建文件/目录

当用户执行 `touch /mnt/test/foo.txt` 时：

1. VFS 路径查找到达 `/mnt/test/`；
2. 发现 `foo.txt` 不存在（`simplefs_lookup` 返回负 dentry）；
3. VFS 调用 `dir->i_op->create()`，即 `simplefs_create()`；
4. 我们分配新 inode，用 `d_instantiate()` 把 dentry 和 inode 关联。

### 读写文件

我们没有自己实现读写——用的是内核提供的 `generic_file_read_iter` / `generic_file_write_iter` + `simple_readpage` / `simple_write_begin` / `simple_write_end`。

这些函数的工作方式：

- **写入**：分配页缓存页面，`copy_from_user()` 到页面，标脏——数据就在内存里了；
- **读取**：在页缓存里找到页面，`copy_to_user()` 返回；
- **没有磁盘回写**：因为 `simple_readpage` 只是清零一个新页面，不从磁盘读。

这正是 ramfs/tmpfs 的工作原理——数据只存在于页缓存。

### 目录遍历

```c
static const struct file_operations simplefs_dir_ops = {
    .iterate_shared = dcache_readdir,
};
```

`dcache_readdir` 直接遍历 dentry 的 `d_subdirs` 列表——因为我们的文件系统完全靠 dcache，不需要自己管理目录块。

## 扩展思路

如果你想在这个基础上继续学习：

| 目标 | 需要实现 |
|------|----------|
| 删除文件 | `inode_operations.unlink` |
| 删除目录 | `inode_operations.rmdir` |
| 重命名 | `inode_operations.rename` |
| 权限修改 | `inode_operations.setattr`（已有 simple 实现） |
| 最大容量限制 | 类似 tmpfs 的 `size=` 挂载选项 |
| 持久化到磁盘 | 改用 `mount_bdev`，实现真正的磁盘读写 |
| 自定义 ioctl | `file_operations.unlocked_ioctl` |
| proc-like 虚拟文件 | 在 `open/read` 里动态生成内容 |

## 与 ramfs/tmpfs 的对比

我们的 simplefs 本质上就是一个简化版 ramfs（源码在 `fs/ramfs/`）。内核的 ramfs 额外做了：

- 更完整的操作（unlink、rename、symlink 等）；
- 正确的安全钩子（LSM）；
- xattr 支持。

tmpfs 则在 ramfs 基础上加了：

- **容量限制**（`size=` 参数）；
- **swap 支持**（内存不足时可以把数据换到 swap 分区）；
- **POSIX 共享内存**（`/dev/shm`）。

## 本章小结

- 一个最小的内核文件系统模块需要实现：`file_system_type`（注册）→ `fill_super`（填超级块）→ inode/dentry/file 的操作表；
- 内存文件系统可以大量复用内核的 `simple_*` / `generic_*` 函数，不到 200 行有效代码就能跑起来；
- `d_instantiate()` 把 dentry 和 inode 关联，`d_make_root()` 创建根 dentry——这是 VFS 的基本操作；
- 理解了这个最小示例，再去读 ramfs、tmpfs、甚至 ext4 的代码，就有了"导航地图"。
